class형 컴포넌트
state 접근 this.state
state 변경 this.setState({})
class MyComstate extends Component {
	constructor(){
		this.state ={
			countNum : 0,
		}
	}
	render() {
		const {countNum} = this.state;
		return(
			<div>
				<h1>{countNum}</h1>
				<button onClick= ()=>{
					this.setState({countNum: countNum+1})
				}>+1</button>
			</div>
		)
	}
}
